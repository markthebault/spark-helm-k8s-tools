1. Get the Spark URL to visit by running these commands in the same shell:
  
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  You can watch the status of by running 'kubectl get svc --namespace {{ .Release.Namespace }} -w {{ template "spark.webui-fullname" . }}'
  
  export SPARK_SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ template "spark.webui-fullname" . }} -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
  echo http://$SPARK_SERVICE_IP:{{ .Values.WebUi.ServicePort }}

2. Run some stuff:
conect to spark master and:
$SPARK_HOME/bin/spark-submit --master spark://{{ template "spark.master-fullname" . }}:{{ .Values.Master.ServicePort }} \
  --deploy-mode cluster \
  --class org.apache.spark.examples.SparkPi \
  ./examples/jars/spark-examples_2.11-2.4.4.jar 10000
